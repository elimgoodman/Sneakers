s = new web.Server {
	port => 8080,
	onRequest => @(req:Request):Response [
		return req.url.match {
			"/" => @(req:Request):Response [
				file_contents = <slurp! "home.ejs">;
				return ErrorResponse.withBody (render file_contents {});
			],
			default => 404
		};
	]
};
